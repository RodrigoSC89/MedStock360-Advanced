@echo off
echo.
echo ====================================
echo       MEDSTOCK360 - INICIANDO...
echo ====================================
echo.
echo Verificando sistema...

REM Verificar se Python está instalado
python --version >nul 2>&1
if errorlevel 1 (
    echo [ERRO] Python nao encontrado!
    echo Execute primeiro: INSTALAR_WINDOWS.bat
    pause
    exit /b 1
)

REM Verificar se Streamlit está instalado
pip show streamlit >nul 2>&1
if errorlevel 1 (
    echo [ERRO] Streamlit nao instalado!
    echo Execute primeiro: INSTALAR_WINDOWS.bat
    pause
    exit /b 1
)

REM Verificar se app.py existe
if not exist "app.py" (
    echo [ERRO] Arquivo app.py nao encontrado!
    echo Certifique-se de estar na pasta correta do projeto.
    pause
    exit /b 1
)

echo [OK] Sistema verificado com sucesso!
echo.
echo Iniciando MedStock360...
echo.
echo ====================================
echo   ACESSE NO NAVEGADOR:
echo   http://localhost:8501
echo.
echo   PRIMEIRO ACESSO:
echo   Usuario: admin
echo   Senha: admin123
echo ====================================
echo.
echo Para parar o sistema, pressione Ctrl+C
echo.

REM Executar Streamlit
streamlit run app.py --server.port 8501 --server.address localhost

echo.
echo Sistema finalizado.
pause
